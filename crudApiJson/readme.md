/*Criar uma tabela de dados*/
CREATE TABLE ANIMALS (
    ID_ANIMALS INT NOT NULL AUTO_INCREMENT,
    DS_NAME VARCHAR(255) NOT NULL,
    NM_AGE INT,
    FL_SEX ENUM('M', 'F'),
    DT_CREATED DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (ID_ANIMALS)
);
